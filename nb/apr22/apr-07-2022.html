        np.expand_dims(_gr_c.run_pi_prune[..., ids, :], -1),
    ], axis=-1)
    boot = bootstrap(
        x=x,
        batch=5,
        dims=dims,
        n_resamples=int(1e4),
    )
    boot = bn.nanmean(bn.nanmean(bn.nanmean(boot, 3), 2), 1)

        if np.isnan(boot[b]).sum():

            XA=boot[b, ..., 0],
            XB=boot[b, ..., 1],
            metric='cosine',

    sim = sim.diagonal(0, 1, 2)
