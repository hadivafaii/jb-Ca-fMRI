def compute_group_adj(mice, mode, task):
    adjacencies, dirs = collect_adjacencies(mice, mode, task)
    nonzero = np.abs(adjacencies).mean(1, keepdims=True) != 0
    a = np.where(nonzero, adjacencies, np.nan)
    a = np.where(np.transpose(nonzero, (0, 2, 1)), a, np.nan)
    a = bn.nanmean(a, 0)
    for i in range(len(a)):
        nans = np.isnan(a[i])
        a[i][nans] = bn.nanmean(a[i])
    a = (a + a.T) / 2
    return a


def do_hier(d, num_clusters, base):
    z = sp_hier.linkage(
        sp_dist.squareform(d),
        method='ward',
        metric='correlation',
        optimal_ordering=True,
    )
    result = get_clust_graph(
        linkage_mat=z,
        labels=base.l2n,
        num_clusters=num_clusters,
    )
    comm_vec = np.zeros(len(base.n2l))
    for c_idx, clist in result['cluster_assigns'].items():
        for lbl in clist:
            comm_vec[base.l2n[lbl]] = c_idx
    comm_vec = comm_vec.astype(int)

    pi_disjoint = np.zeros((num_clusters, len(comm_vec)))
    for i, c in enumerate(comm_vec):
        pi_disjoint[c - 1, i] = 1

    mapping = base.align_centroid_to_structs(pi_disjoint)
    pi_disjoint = pi_disjoint[mapping]

    partition = np.argmax(pi_disjoint, 0) + 1
    proj = base.project_vec(partition, add_disjoint=False)
    
    return partition, proj


def show_hier(
        tr_1: dict = None,
        tr_2: dict = None,
        labels: Tuple[str, str] = ('BOLD-Lite', r"Ca$^{2+}$"),
        display: bool = True,
        **kwargs, ):
    defaults = {
        'fig_y': 2.0,
        'fig_x': 10.0,
        'w_ratio': 1.6,
        'h_ratio': 1.0,
        'cmap_anat': 'bone',
        'interpolation': 'none',
    }
    for k, v in defaults.items():
        if k not in kwargs:
            kwargs[k] = v

    assert list(tr_1) == list(tr_2)
    n_clusters = list(tr_1)
            
    ncols = 4
    nrows = len(n_clusters)
    fig, axes = create_figure(
        nrows=nrows, ncols=ncols,
        figsize=(kwargs['fig_x'], kwargs['fig_y'] * nrows),
        gridspec_kw={
            'width_ratios': [kwargs['w_ratio'], 1, 1, kwargs['w_ratio']],
            'height_ratios': [1] * (nrows - 1) + [kwargs['h_ratio']]},
        style='white',
    )
    for i, k in enumerate(n_clusters):
        cmap = get_cluster_cmap(range(k))

        x_top = tr_1[k]['proj-top'][0]
        x_flat = tr_1[k]['proj-flat'][0]

        axes[i, 0].imshow(
            tr_1[k]['flat'], cmap=kwargs['cmap_anat'])
        x2plt = np.ma.masked_where(
            x_flat == 0, x_flat)
        axes[i, 0].imshow(
            x2plt, cmap=cmap, interpolation=kwargs['interpolation'])
        axes[i, 0].set_ylabel(f"Clust_{k}\n", fontsize=13)

        x2plt = np.ma.masked_where(
            tr_1[k]['top'] == 0, tr_1[k]['top'])
        axes[i, 1].imshow(
            x2plt, cmap=kwargs['cmap_anat'])
        x2plt = np.ma.masked_where(
            x_top == 0, x_top)
        axes[i, 1].imshow(
            x2plt, cmap=cmap, interpolation=kwargs['interpolation'])

        x_top = tr_2[k]['proj-top'][0]
        x_flat = tr_2[k]['proj-flat'][0]
        if i == nrows - 1:
            interp = 'none'
        else:
            interp = kwargs['interpolation']

        x2plt = np.ma.masked_where(
            tr_2[k]['top'] == 0, tr_2[k]['top'])
        axes[i, 2].imshow(
            x2plt, cmap=kwargs['cmap_anat'])
        x2plt = np.ma.masked_where(
            x_top == 0, x_top)
        axes[i, 2].imshow(
            x2plt, alpha=0.9, cmap=cmap, interpolation=kwargs['interpolation'])

        axes[i, 3].imshow(
            tr_2[k]['flat'], cmap=kwargs['cmap_anat'])
        x2plt = np.ma.masked_where(
            x_flat == 0, x_flat)
        axes[i, 3].imshow(
            x2plt, cmap=cmap, interpolation=kwargs['interpolation'])

        if i == 0:
            axes[i, 0].set_title(f"{labels[0]} (flat)", y=1.04)
            axes[i, 1].set_title(f"{labels[0]} (top)", y=1.06)
            axes[i, 2].set_title(f"{labels[1]} (top)", y=1.06)
            axes[i, 3].set_title(f"{labels[1]} (flat)", y=1.04)

    remove_ticks(axes)
    if display:
        plt.show()
    else:
        plt.close()
    return fig, axes
