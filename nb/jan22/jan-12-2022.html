def low_tsnr_mask(x: np.ndarray, p: float = 5.0, axis: int = -1):
    tsnr = x.mean(axis) / x.std(axis)
    vals = tsnr[np.isfinite(tsnr)]
    mask = tsnr > np.percentile(vals, p)
    return mask, tsnr
