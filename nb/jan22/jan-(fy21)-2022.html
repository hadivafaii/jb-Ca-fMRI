dlist = []
for row in df.iterrows():
    yr = int(row[1]['Fiscal Year'])
    if yr == 21:
        continue
    data = {
        'Total': float(row[1]['Total Revenue']),
        'External': float(row[1]['Externally Funded']),
        'UMD': float(row[1]['UMD Funded']),
        'Other': float(row[1]['Other Agency Funded*'].replace(',', '')),
    }
    dlist.append({
        'Year': [yr] * len(data),
        'Funding': data.keys(),
        'Revenue': data.values(),
    })
